SELECT CONCAT(e.nom, ' ', e.prenom) nomComplet, s.nom serv
FROM employe e
JOIN service S
USING(idService)
ORDER BY nomComplet;

4.
SELECT CONCAT(e.nom, ' ', e.prenom) AS nomComplet, s.nom AS serv
FROM employe e
LEFT JOIN service s
ON e.idService = s.idService
WHERE s.idService IS NULL
ORDER BY nomComplet;
